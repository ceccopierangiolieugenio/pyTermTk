        forceColor = self.color!=TTkColor.RST

    def textUpdated(self, text):
        w, h = self.size()
        if w<len(text) or h<1:
            self.resize(len(text),1)
        self.setMinimumSize(len(text), 1)
        self.update()

    def colorUpdated(self, color):
        self.update()
